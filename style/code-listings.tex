\definecolor{codehighlight}{rgb}{0.827,0.322,0.07}

\newcommand{\lstinputlistinghl}[3][]{%
  \lstinputlisting[%
    belowskip=0pt,
    lastline=\number\numexpr#2-1\relax,
    showlines=true,
    #1
  ]{#3}%
  \lstinputlisting[%
    aboveskip=0pt,
    belowskip=0pt,
    backgroundcolor=\color{codehighlight!15},
    firstline=#2,
    lastline=#2
  ]{#3}
  \lstinputlisting[%
    aboveskip=0pt,
    firstline=\number\numexpr#2+1\relax
  ]{#3}
}
